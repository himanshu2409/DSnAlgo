#include <bits/stdc++.h>
using namespace std;
void rotate(vector<vector<int> > &A) {
    int i,j,t;
    for(i=0;i<A.size();i++)
    {
        for(j=i;j<A[i].size();j++)
        {
            t=A[i][j];
            A[i][j]=A[j][i];
            A[j][i]=t;
        }
    }
    i=0;j=A[0].size()-1;
    while(i<j)
    {
        t=0;
        while(t<A.size())
        {
            int temp=A[t][i];
            A[t][i]=A[t][j];
            A[t][j]=temp;
            t++;
        }
        i++;
        j--;
    }
}
int main() {
    vector<vector<int>> vec{ { 1, 2, 3 },
                              { 4, 5, 6 },
                              { 7, 8, 9 } };
           rotate(vec);        

     for (int i = 0; i < 3; i++) {
        for (auto it = vec[i].begin();it != vec[i].end(); it++)
            cout << *it << " ";
        cout << endl; 
     }
     return 0;          
}
